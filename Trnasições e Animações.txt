function navegar(destino) {
    const transicao = document.getElementById('transicao');
    
    somClique.volume = document.getElementById('volumeEfeitos').value;
    somClique.play();

    if (destino === 'menu') {
        iniciarMusica();
    }

    transicao.style.width = '200vw';
    transicao.style.height = '200vw';

    setTimeout(() => {
        document.querySelectorAll('.tela').forEach(tela => {
            tela.classList.remove('ativa');
        });
        document.getElementById(destino).classList.add('ativa');

        transicao.style.width = '0';
        transicao.style.height = '0';
    }, 400);
}

function voltarInicio() {
    navegar('inicio');
}

function voltarMenu() {
    navegar('menu');
}
